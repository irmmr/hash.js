Spa : single page application
//create new object by spa
var apExample = new Hash.spa();

//use obj -> app for manage a spa
apExample.app({
	el : 'app',
	def : 'home',
	component : {
		'home' : {
			title : 'home (welcome) | spa app',
			main : 'welcome to home!'
		},
		'home/start' : {
			title : 'home start | spa app',
			main : 'start working with us',
		 	do : function() {
				alert('start now!');
		 	}
	  	}
	},
	error : {
		'404' : {
			main : 'page not found!',
			title : 'error 404 | spa app',
		}
	}
});






Router
//create new object by spa
var apExample = new Hash.spa();
	
//use obj -> app for manage a router
apExample.router({
	router : 'page/{any}',
	do : function(e) {
		alert(e[0]);
	}
});
