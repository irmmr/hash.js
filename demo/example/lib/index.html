Check page hash with have
var exa = new Hash.lib();
/*
  have(), this check if page have any hash
  have('test'), this check if page have 'test' in hash
*/
exa.have();  //returned hash status : boolean

//example
if (exa.have()) {
  //if page have hash : true else : false
}
//Or
if (exa.have(`hello`)) {
  //if page hash have 'hello' string : true else : false
}





Clear page hash
var exa = new Hash.lib();
/* you can clear hash with '#' or without it */
exa.clear();  //returned hash status : boolean

//clear in def
exa.clear();
//Or with setting
exa.clear({
  sharp : false  //clear with sharp or not?
});





Remove from page hash
var exa = new Hash.lib();
/* you can remove some strings in hash */

// Your hash first: #hello-baby-get-out
exa.remove({
  words : [
    `hello`, `out`, `baby`
  ]
});
// Your hash after: #--get-





Lock page hash
var exa = new Hash.lib();
/* by this user can't change first hash */
exa.lock();
